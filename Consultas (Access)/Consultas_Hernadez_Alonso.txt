7)--Nombre Completo de los Clientes con Cuentas y su Saldo a Favor
	SELECT  Nombre, SUM(Saldo) AS Saldo_a_Favor 
	FROM  Cliente RIGHT JOIN Cuenta 
	ON Cliente.IdCliente = Cuenta.IdCliente 
	GROUP BY Cliente.IdCliente, Cliente.Nombre

8)--El Saldo Total por Cliente (Nombre Completo, SaldoTotal)
	SELECT  Nombre, SUM(Cuenta.Saldo - Prestamo.Monto) AS Saldo_a_Favor 
	FROM  Cliente RIGHT JOIN Cuenta
	ON Cliente.IdCliente = Cuenta.IdCliente   
	GROUP BY Cliente.IdCliente, Cliente.Nombre, Prestamo.IdCliente HAVING Prestamo.IdCliente = Cliente.IdCliente 

	SELECT SUM(Cuenta.Saldo - Prestamo.Monto) AS Saldo_a_Favor
	FROM(
		(Cliente INNER JOIN Cuenta ON Cliente.IdCliente = Cuenta.IdCliente	)
				INNER JOIN Prestamo ON Prestamo.IdCliente = Cliente.IdCliente)

9)--Números de los Préstamos y Montos que sean Inferiores o iguales a 1000
	SELECT TOP 1 
	( SELECT COUNT(*) FROM Cuenta WHERE Saldo<=1000) AS nSaldo, 
    ( SELECT COUNT(*) FROM Prestamo WHERE Monto<=1000) AS nMonto 
	FROM Cuenta,Prestamo
